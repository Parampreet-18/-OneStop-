// Function to display hotels dynamically
function displayHotels() {
    const container = document.getElementById("hotelsContainer");
    container.innerHTML = ""; // Clear previous hotels

    hotels.forEach((hotel) => {
        let hotelDiv = document.createElement("div");
        hotelDiv.classList.add("hotel");
        hotelDiv.innerHTML = `
            <img src="${hotel.image}" alt="${hotel.name}">
            <h2>${hotel.name}</h2>
            <p>${hotel.description}</p>
            <p class="price">${hotel.price}</p>
            <button class="book-now" onclick="bookNow('${hotel.name}')">Book Now</button>
        `;
        container.appendChild(hotelDiv);
    });
}

// Function to handle booking
function bookNow(hotelName) {
    alert(`Your booking for ${hotelName} is confirmed! ðŸŽ‰`);
}

// Function for searching hotels
function searchHotels() {
    let searchValue = document.getElementById("searchBox").value.toLowerCase();
    let filteredHotels = hotels.filter(hotel => hotel.name.toLowerCase().includes(searchValue));
    
    const container = document.getElementById("hotelsContainer");
    container.innerHTML = ""; // Clear previous hotels

    if (filteredHotels.length === 0) {
        container.innerHTML = "<p style='text-align:center; font-size:18px;'>No results found ðŸ˜”</p>";
    } else {
        filteredHotels.forEach((hotel) => {
            let hotelDiv = document.createElement("div");
            hotelDiv.classList.add("hotel");
            hotelDiv.innerHTML = `
                <img src="${hotel.image}" alt="${hotel.name}">
                <h2>${hotel.name}</h2>
                <p>${hotel.description}</p>
                <p class="price">${hotel.price}</p>
                <button class="book-now" onclick="bookNow('${hotel.name}')">Book Now</button>
            `;
            container.appendChild(hotelDiv);
        });
    }
}

// Call function to display hotels on page load
window.onload = displayHotels;
